Program 12:Array Of Strings

package PracticalProjects;

public class ArrayOfStrings {
	 public static void main(String[] args) {  
         String[] str = { "Priya", "Anshu", "Parnika" };  
         boolean x = false; 
         int n = 0;
         String s = "Anshu"; 
         for (int i = 0; i < str.length; i++) {  
             if(s.equals(str[i])) {  
                 n = i;
                 break;  
             }  
         }  
         if(x= true)  
             System.out.println(s +" String is found at index "+n);  
         else  
             System.out.println(s +" String is not found in the array");  
     }  

}

Output:

Anshu String is found at index 1
