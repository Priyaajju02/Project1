Program 5:Constructor

package PracticalProjects;

public class Constructors {
	int id;  
    String name;  
    Constructors(int i,String s){  
    id =i;  
    name =s;  
    }   
    void display(){
    	System.out.println(id+" "+name);
    	}  
   
    public static void main(String args[]){    
    Constructors a = new Constructors(1,"Alexa"); 
    Constructors b = new Constructors(2,"Anshu");   
    a.display();  
    b.display();  
   }  

}


Output:

1 Alexa
2 Anshu
