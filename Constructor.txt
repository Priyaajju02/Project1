Program 5:Constructor
package PracticalProjects;

import java.lang.reflect.Constructor;

public class Constructors {
	int id;  
    String name;  
    Constructors(int i,String n){  
    id = i;  
    name = n;  
    }   
    void display(){
    	System.out.println(id+" "+name);
    	}  
   
    public static void main(String args[]){    
    Constructors c1 = new Constructors(1,"Anshu"); 
    Constructors c2 = new Constructors(2,"Alexa");   
    c1.display();  
    c2.display();  
   }  

}




Output:

1 Anshu
2 Alexa
